JavaScript Approach (For complex interactions):

function typeWriter(element, text, speed = 50) {
  let i = 0;
  element.innerHTML = '';
  
  function type() {
    if (i < text.length) {
      element.innerHTML += text.charAt(i);
      i++;
      setTimeout(type, speed + Math.random() * 50);
    }
  }
  
  type();
}
